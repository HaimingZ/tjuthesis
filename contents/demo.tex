如果表格需要注释，则需要使用 threeparttable 包，结果如表~\ref{tab:2} 所示。
\begin{table}[thb]
    \centering\small
    \caption{一个包含注释的三线表}\label{tab:2}
    \begin{threeparttable}
        \begin{tabular}{L{3cm}C{3cm}R{3cm}}
            \toprule
            表头 & 表头 & 表头 \\
            \midrule
            表格内容\tnote{*} & 表格内容 & 表格内容 \\
            表格内容\tnote{**} & 表格内容 & 表格内容 \\
            表格内容\tnote{***} & 表格内容 & 表格内容 \\
            ... & ... & ... \\
            ... & ... & ... \\
            ... & ... & ... \\
            ... & ... & ... \\
            ... & ... & ... \\
            \bottomrule
        \end{tabular}
        \begin{tablenotes}
            \item [*] 这是一个注释。
            \item [**] This is a table note.
            \item [***] 它们都在表格下方。
        \end{tablenotes} 
    \end{threeparttable}
\end{table}
